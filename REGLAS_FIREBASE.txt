// --- REGLAS DE FIRESTORE (BASE DE DATOS) ---
// Copia esto en: Firestore Database > Reglas

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Regla para la App "Casa La Martina"
    // Permite leer y escribir SÓLO si el usuario está conectado (el código usa Auth Anónimo)
    match /artifacts/casa-la-martina-app/{document=**} {
      allow read, write: if request.auth != null;
    }

  }
}

// -----------------------------------------------------

// --- REGLAS DE STORAGE (IMÁGENES/ARCHIVOS) ---
// Copia esto en: Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Permite subir y ver archivos si el usuario está conectado
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
    
  }
}
